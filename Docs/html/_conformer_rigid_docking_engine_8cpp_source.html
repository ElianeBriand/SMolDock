124  void ConformerRigidDockingEngine::runDockingEngine() {
125  using namespace boost::accumulators;
126  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_score;
127  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_duration;
128  std::vector<double> scores;
129  scores.reserve(this->conformer_num);
130 
131  record_timings(begin_docking);
132 
133  for(auto& conformer : this->viConformers)
134  {
135  record_timings(begin_docking_this_conformer);
136 
137  iTransform transform = iTransformZeroInit();
138  iGradient translationGradient;
139  double score = Score::basic_scoring_func(conformer,transform, this->protein);
140  double oldscore;
141  double score_progress = 100.0;
142  double differential_epsilon = 0.01;
143  unsigned int iterationNbr = 0;
144 
145  while(score_progress > 0.001 )
146  {
147  oldscore = score;
148 
149  // Compute approx gradient to find in which direction to go
150  iTransform transform_dx = transform;
151  transform_dx.transl.x += differential_epsilon;
152  translationGradient.dx = oldscore - Score::basic_scoring_func(conformer,transform_dx, this->protein);
153 
154  iTransform transform_dy = transform;
155  transform_dy.transl.y += differential_epsilon;
156  translationGradient.dy = oldscore - Score::basic_scoring_func(conformer,transform_dy, this->protein);
157 
158  iTransform transform_dz = transform;
159  transform_dz.transl.z += differential_epsilon;
160  translationGradient.dz = oldscore - Score::basic_scoring_func(conformer,transform_dz, this->protein);
161 
162 
163  // Now we want to know how far in that direction
164  // We do a line-search
165 
166  iTransform transform_linesearch;
167 
168  double step = 0.0001;
169  double previous_step = 0.0;
170  double temp_score;
171  double previous_temp_score = std::numeric_limits<double>::max();
172 
173  // We go further and further in the direction of the gradient, until we are not bettering the score
174  while(true) {
175  transform_linesearch = transform;
176  transform_linesearch.transl.x += step * translationGradient.dx;
177  transform_linesearch.transl.y += step * translationGradient.dy;
178  transform_linesearch.transl.z += step * translationGradient.dz;
179 
180  temp_score = Score::basic_scoring_func(conformer, transform_linesearch, this->protein);
181 
182  double diff_oldscore_tempscore = oldscore - temp_score;
183  double diff_previous_tempscore = previous_temp_score - temp_score;
184 
185  previous_temp_score = temp_score;
186 
187  if (diff_oldscore_tempscore > 0 && // temp_score is lower than the old score thus better
188  diff_previous_tempscore > 0 ) // It is also lower than in the previous iteration
189  {
190  // so we jump further to see if it can be even better
191  previous_step = step;
192  step = step * 10;
193  continue;
194  } else {
195  // We hit a step that is not better than either oldscore or the previous iterator
196  break;
197  }
198  }
199 
200 
201  // We then go back to previous_step for updating the score
202 
203  if(previous_step == 0.0)
204  {
205  // This means we could not do any jump forward
206  // So this is the best we can do (to the precision of the starting step 0.0001 (angstrom))
207  // So we stop the search (this is one of the stop condition, the other is score_progress)
208  break;
209  }
210 
211  // Otherwise we update the transform and go for another round
212 
213  transform.transl.x += previous_step * translationGradient.dx;
214  transform.transl.y += previous_step * translationGradient.dy;
215  transform.transl.z += previous_step * translationGradient.dz;
216 
217  score = Score::basic_scoring_func(conformer,transform, this->protein);
218  score_progress = oldscore - score;
219 
220  iterationNbr++;
221 
222  //BOOST_LOG_TRIVIAL(debug) << " Score: " << score << " (progress: "<< score_progress << " )";
223 
224  }
225 
226  scores.push_back(score);
227 
228  record_timings(end_docking_this_conformer);
229 
230  acc_score(score);
231  acc_duration(
232  static_cast< std::chrono::duration<double> >(end_docking_this_conformer - begin_docking_this_conformer).count()
233  );
234 
235  }
236 
237  record_timings(end_docking);
238 
239 #ifdef SMOLDOCK_VERBOSE_DEBUG
240  auto total_docking_duration = static_cast< std::chrono::duration<double> >(end_docking - begin_docking).count();
241 
242  BOOST_LOG_TRIVIAL(debug) << "Timings ConformerRigidDockingEngine docking runs"
243  << "\n TOTAL: " << total_docking_duration << "s"
244  << "\n Mean per conformer: " << mean(acc_duration) << "s"
245  << "\n StdDev per conformer: " << std::sqrt(moment<2>(acc_duration)) << "s";
246 #endif
247 
248  BOOST_LOG_TRIVIAL(info) << "Results:";
249  BOOST_LOG_TRIVIAL(info) << " Score Mean: " << mean(acc_score);
250  BOOST_LOG_TRIVIAL(info) << " Score StdDev: " << std::sqrt(moment<2>(acc_score));
251  BOOST_LOG_TRIVIAL(info) << " Scores: ";
252  for(auto& score: scores)
253  {
254  BOOST_LOG_TRIVIAL(info) << " " << score;
255  }
256

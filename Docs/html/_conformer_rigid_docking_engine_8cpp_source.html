137  void ConformerRigidDockingEngine::runDockingEngine() {
138  using namespace boost::accumulators;
139  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_score;
140  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_duration;
141  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_iteration;
142 
143 
144 
145 
146  record_timings(begin_docking);
147 
148  for(auto& conformer : this->viConformers)
149  {
150 
151  BOOST_LOG_TRIVIAL(debug) << "Initial score : " << Score::vina_like_rigid_inter_scoring_func(conformer,iTransformIdentityInit(),this->protein);
152  continue;
153 
154  record_timings(begin_docking_this_conformer);
155 
156  GradientDescentLineSearch gradOptimizer(Score::vina_like_rigid_inter_scoring_func);
157  gradOptimizer.setProtein(&(this->protein));
158  gradOptimizer.setStartingConformer(&conformer);
159 
160  gradOptimizer.optimize();
161 
162  double score = gradOptimizer.getScore();
163  iConformer result = gradOptimizer.getFinalConformer();
164 
165 
166  this->scores.push_back(score);
167  this->final_iConformer.push_back(result);
168 
169  record_timings(end_docking_this_conformer);
170 
171  acc_iteration(gradOptimizer.getIterationNumber());
172  acc_score(score);
173  acc_duration(
174  static_cast< std::chrono::duration<double> >(end_docking_this_conformer - begin_docking_this_conformer).count()
175  );
176 
177  }
178 
179  record_timings(end_docking);
180 
181 #ifdef SMOLDOCK_VERBOSE_DEBUG
182  auto total_docking_duration = static_cast< std::chrono::duration<double> >(end_docking - begin_docking).count();
183 
184  BOOST_LOG_TRIVIAL(debug) << "Timings ConformerRigidDockingEngine docking runs"
185  << "\n TOTAL: " << total_docking_duration << "s"
186  << "\n Mean per conformer: " << mean(acc_duration) << "s"
187  << "\n StdDev per conformer: " << std::sqrt(moment<2>(acc_duration)) << "s";
188 #endif
189 
190  BOOST_LOG_TRIVIAL(info) << "Results:";
191  BOOST_LOG_TRIVIAL(info) << " Iteration Mean: " << mean(acc_iteration);
192  BOOST_LOG_TRIVIAL(info) << " Iteration StdDev: " << std::sqrt(moment<2>(acc_iteration));
193  BOOST_LOG_TRIVIAL(info) << " Score Mean: " << mean(acc_score);
194  BOOST_LOG_TRIVIAL(info) << " Score StdDev: " << std::sqrt(moment<2>(acc_score));
195  BOOST_LOG_TRIVIAL(info) << " Scores: ";
196  for(auto& score: scores)
197  {
198  BOOST_LOG_TRIVIAL(info) << " " << score;
199  }

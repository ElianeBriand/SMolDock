47  std::string atomTypeToString(const Atom::AtomType t) {
48  auto it = std::find_if(Atom::AtomTypeLabel.begin(), Atom::AtomTypeLabel.end(),
49  [&](const std::tuple<Atom::AtomType, std::string, std::string, double> &e) {
50  return std::get<0>(e) == t;
51  });
52  if (it != Atom::AtomTypeLabel.end()) {
53  return std::get<1>(*it);
54  }else{
55  BOOST_LOG_TRIVIAL(error) << "Encountered unknown atom AtomType while converting to string: " << static_cast<unsigned char>(t);
56  return std::string("");
57  }

60  Atom::AtomType stringToAtomType(const std::string &symbol_or_name) {
61  auto it = std::find_if(Atom::AtomTypeLabel.begin(), Atom::AtomTypeLabel.end(),
62  [&](const std::tuple<Atom::AtomType, std::string, std::string, double> &e) {
63  return (std::get<1>(e) == symbol_or_name) || (std::get<2>(e) == symbol_or_name);
64  });
65  if (it != Atom::AtomTypeLabel.end()) {
66  return std::get<0>(*it);
67  } else {
68  BOOST_LOG_TRIVIAL(error) << "Encountered unknown atom symbol_or_name while converting from string: " << symbol_or_name;
69  return Atom::AtomType::unknown;
70  }
71  }

118  Atom::Atom(const std::string &symbol_or_name, bool PDBFormat, AminoAcid::AAType resType) {
119  if (PDBFormat) {
120  auto first_letter = symbol_or_name.substr(0, 1); // One letter code for atom is always same as usual
121  this->type = stringToAtomType(first_letter);
122  this->atomClassInResidue = symbol_or_name;
123  fromResidue = true;
124  } else {
125  this->type = stringToAtomType(symbol_or_name);
126  }
127  this->residueType = resType; // Default : heteroatom
128  this->AtomID = nextAtomID;
129  this->atomicRadius = atomTypeToAtomicRadius(this->type);
130  nextAtomID++;
131  }

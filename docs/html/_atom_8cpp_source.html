48  auto it = std::find_if(Atom::AtomTypeLabel.begin(), Atom::AtomTypeLabel.end(),
49  [&](const std::tuple<Atom::AtomType, std::string, std::string, double> &e) {

77  [&](const std::tuple<Atom::AtomType, std::string, std::string, double> &e) {

141  Atom::Atom(const std::string &symbol_or_name, bool PDBFormat, AminoAcid::AAType resType) {
142  if (PDBFormat) {
143  auto first_letter = symbol_or_name.substr(0, 1); // One letter code for atom is always same as usual
144  this->type = stringToAtomType(first_letter);
145  this->atomClassInResidue = symbol_or_name;
146  fromResidue = true;
147  } else {
148  this->type = stringToAtomType(symbol_or_name);
149  }
150  this->residueType = resType; // Default : heteroatom
151  this->AtomID = nextAtomID;
152  this->atomicRadius = atomTypeToAtomicRadius(this->type);
153  nextAtomID++;
154  }

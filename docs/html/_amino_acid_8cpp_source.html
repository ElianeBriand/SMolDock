65  std::string resTypeToString(AminoAcid::AAType t) {
66  auto it = std::find_if(AminoAcid::AAShorthandSet.begin(), AminoAcid::AAShorthandSet.end(),
67  [&](const std::tuple<AminoAcid::AAType, std::string, std::string> &e) {

70  if (it != AminoAcid::AAShorthandSet.end()) {

77  AminoAcid::AAType stringToResType(const std::string &shorthand_or_name) {
78  auto it = std::find_if(AminoAcid::AAShorthandSet.begin(), AminoAcid::AAShorthandSet.end(),
79  [&](const std::tuple<AminoAcid::AAType, std::string, std::string> &e) {
80  return (std::get<1>(e) == shorthand_or_name) ||
81  (std::get<2>(e) == shorthand_or_name);
82  });
83  if (it != AminoAcid::AAShorthandSet.end()) {
84  return std::get<0>(*it);
85  } else {
86 #ifdef SMOLDOCK_VERBOSE_DEBUG
87  std::cout << "[!] Encountered unknown residue shorthand_or_name : " << shorthand_or_name << std::endl;
88 #endif

109  {

111  if(skipHydrogen && atom->getAtomType() == Atom::AtomType::hydrogen)
112  continue;

114  prot.type.push_back(atom->getAtomicNumber());

120  prot.y.push_back(std::get<1>(pos));
121  prot.z.push_back(std::get<2>(pos));

123  prot.atomicRadius.push_back(atom->getAtomicRadius());

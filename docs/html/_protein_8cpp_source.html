36  bool Protein::populateFromPDB(const std::string &filename) {
37 
38 
39 
40  // Read all non-water atoms/stuff and water in two separate system
41  // TODO : do something with the water or switch to one system
42  ESBTL::PDB_line_selector_two_systems sel;
43 
44  std::vector<ESBTL::Default_system> systems;
45 
46 
47  ESBTL::All_atom_system_builder<ESBTL::Default_system> builder(systems, sel.max_nb_systems());
48 
49 
50  if (ESBTL::read_a_pdb_file(filename, sel, builder,
51  ESBTL::Accept_none_occupancy_policy<ESBTL::PDB::Line_format<> >())) {
52 
53  if (systems.empty() || systems[0].has_no_model()) {
54  BOOST_LOG_TRIVIAL(error) << "No atoms found in protein PDB";
55  std::exit(-1);
56  }
57 
58  /*
59  * For reference purpose, the hierarchy of the PDB file reading is :
60  * System (defined by ESBTL, like input filter in some ways)
61  * Model (PDB-file specification : more or less, protein)
62  * Chain (Amino acid chain)
63  * Residue (=Amino acid)
64  * Atom
65  *
66  * Heteroatoms (Cl, or crystallographic ions) are mixed in at the residue level.
67  * We detect those if residueName is not in {set of defined residueName}
68  * TODO : Differentiate between legit biological ions (metalloprotein, ...) and crystallographic heavy atoms
69  */
70 
71  unsigned int total_nb_model = 0, total_nb_chain = 0, total_nb_atom = 0, total_nb_hetatm = 0, total_nb_residue = 0;
72 
73  using namespace boost::accumulators;
74  accumulator_set<double, stats<tag::mean, tag::count > > acc_x, acc_y, acc_z;
75 
76 
77  //Iterate over all models in the first system
78  for (ESBTL::Default_system::Models_iterator
79  it_model = systems[0].models_begin();
80  it_model != systems[0].models_end();
81  ++it_model) {
82 
83  const ESBTL::Default_system::Model &model = *it_model;
84  total_nb_model++;
85 
86  // Iterating over chains
87  for (ESBTL::Default_system::Model::Chains_const_iterator it_ch = model.chains_begin();
88  it_ch != model.chains_end(); ++it_ch) {
89 
90  total_nb_chain++;
91 
92  // Iterating over residues in the chain
93  for (ESBTL::Default_system::Chain::Residues_const_iterator it_res = it_ch->residues_begin();
94  it_res != it_ch->residues_end(); ++it_res) {
95  total_nb_residue++;
96 
97  if (stringToResType(it_res->residue_name()) == AminoAcid::AAType::unknown) {
98  // We have an heteroatom
99  this->heteroatoms.emplace_back(std::make_shared<Atom>(it_res->residue_name()));
100  total_nb_hetatm++;
101  total_nb_atom++;
102  continue;
103  }
104 
105  // else : It's a normal residue
106  auto current_residue = this->aminoacids.emplace_back(
107  std::shared_ptr<AminoAcid>(new AminoAcid(it_res->residue_name()))
108  );
109 
110  current_residue->setAAId(total_nb_residue);
111 
112  // Iterate over atoms in the residue
113  for (ESBTL::Default_system::Residue::Atoms_const_iterator it_atm = it_res->atoms_begin();
114  it_atm != it_res->atoms_end(); ++it_atm) {
115 
116  assert(it_atm->is_hetatm() == 0); // We expect heteroatoms to have been taken care of previously
117 
118  total_nb_atom++;
119 
120  auto current_atom = current_residue->atoms.emplace_back(
121  std::shared_ptr<Atom>(new Atom(it_atm->atom_name(), true,
122  stringToResType(it_res->residue_name())))
123  );
124 
125  current_atom->setAtomPosition(std::make_tuple(it_atm->x(), it_atm->y(), it_atm->z()));
126 
127  acc_x(it_atm->x());
128  acc_y(it_atm->y());
129  acc_z(it_atm->z());
130 
131  }
132  }
133  }
134 
135  this->center_x = mean(acc_x);
136  this->center_y = mean(acc_y);
137  this->center_z = mean(acc_z);
138 
139 
140  BOOST_LOG_TRIVIAL(info) << "Loaded protein : " << filename << "\n --> " << total_nb_model << " models, "
141  << total_nb_chain << " chains, " << total_nb_residue << " residues, "
142  << total_nb_atom << " atoms (" << total_nb_hetatm << " heteroatoms)";
143 
144 #ifdef SMOLDOCK_VERBOSE_DEBUG
145  BOOST_LOG_TRIVIAL(debug) << "Protein centered on : ("
146  << this->center_x << ", " << this->center_y << ", " << this->center_z << ") [n=" <<
147  count(acc_x) << "]";
148 #endif
149 
150  }
151  } else {
152  BOOST_LOG_TRIVIAL(error) << "Could not load protein PDB file: " << filename ;
153  return false;
154  }
155 
156  return true;
157  }

159  iProtein Protein::getiProtein() {
160  /*
161  * struct iProtein {
163  double center_x,center_y,center_z;
164  unsigned long size;
165  std::vector<double> x,y,z;
166  std::vector<unsigned char> type;
168  std::map<unsigned int, std::tuple<unsigned long,unsigned long> > AAId_to_AtomPositionInVect;
169  };
170  */
171  iProtein prot;
172 
173  prot.center_x = this->center_x;
174  prot.center_y = this->center_y;
175  prot.center_z = this->center_z;
176 
177 
178  for(auto& residue: this->aminoacids)
179  {
180  unsigned long size_before = prot.x.size();
181  residue->filliProtein(prot);
182  unsigned long size_after = prot.x.size();
183 
184  // Visual demo :
185  //
186  // index
187  //
188  // size 0
189  //
190  // index 0 1
191  // [A] [A] For residue A : <begin_idx, end_idx> = <0,1> = <size_before, size_after -1>
192  // size 0 > 1 > 2
193  //
194  // index 0 1 2 3 4
195  // [A] [A] [B] [B] [B] For residue B : <begin_idx, end_idx> = <2,4> = <size_before, size_after -1>
196  // size 0 > 1 > 2 > 3 > 4 > 5
197  //
198  // The size_before gives us the index of the first atom that belongs to this residue
199  // The size_after lags behind by one.
200  // This works even if there is only one atom added (then begin_idx = end_idx = size_before = size_after-1)
201  // However there is an edge case : when the residue has no atom, this is incorrect, so :
202  assert(size_after > size_before); // At least one atom added
203 
204  prot.AAId_to_AtomPositionInVect[residue->getAAId()] = std::make_tuple(size_before, size_after - 1);
205  }
206 
207  return prot;
208  }

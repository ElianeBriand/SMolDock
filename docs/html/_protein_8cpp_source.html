41  bool Protein::populateFromPDB(const std::string &filename, std::vector< std::shared_ptr<InputPostProcessor::InputPostProcessor> > postProcessors) {
42 
43 
44 
45  // Read all non-water atoms/stuff and water in two separate system
46  // TODO : do something with the water or switch to one system
47  ESBTL::PDB_line_selector_two_systems sel;
48 
49  std::vector<ESBTL::Default_system> systems;
50 
51 
52  ESBTL::All_atom_system_builder<ESBTL::Default_system> builder(systems, sel.max_nb_systems());
53 
54 
55  if (ESBTL::read_a_pdb_file(filename, sel, builder,
56  ESBTL::Accept_none_occupancy_policy<ESBTL::PDB::Line_format<> >())) {
57 
58  if (systems.empty() || systems[0].has_no_model()) {
59  BOOST_LOG_TRIVIAL(error) << "No atoms found in protein PDB";
60  std::exit(-1);
61  }
62 
63  /*
64  * For reference purpose, the hierarchy of the PDB file reading is :
65  * System (defined by ESBTL, like input filter in some ways)
66  * Model (PDB-file specification : more or less, protein)
67  * Chain (Amino acid chain)
68  * Residue (=Amino acid)
69  * Atom
70  *
71  * Heteroatoms (Cl, or crystallographic ions) are mixed in at the residue level.
72  * We detect those if residueName is not in {set of defined residueName}
73  * TODO : Differentiate between legit biological ions (metalloprotein, ...) and crystallographic heavy atoms
74  */
75 
76  unsigned int total_nb_model = 0, total_nb_chain = 0, total_nb_atom = 0, total_nb_hetatm = 0, total_nb_residue = 0;
77 
78  using namespace boost::accumulators;
79  accumulator_set<double, stats<tag::mean, tag::count, tag::min, tag::max > > acc_x, acc_y, acc_z;
80 
81 
82  //Iterate over all models in the first system
83  for (ESBTL::Default_system::Models_iterator
84  it_model = systems[0].models_begin();
85  it_model != systems[0].models_end();
86  ++it_model) {
87 
88  const ESBTL::Default_system::Model &model = *it_model;
89  total_nb_model++;
90 
91  // Iterating over chains
92  for (ESBTL::Default_system::Model::Chains_const_iterator it_ch = model.chains_begin();
93  it_ch != model.chains_end(); ++it_ch) {
94 
95  total_nb_chain++;
96 
97  // Iterating over residues in the chain
98  for (ESBTL::Default_system::Chain::Residues_const_iterator it_res = it_ch->residues_begin();
99  it_res != it_ch->residues_end(); ++it_res) {
100  total_nb_residue++;
101 
102  if (stringToResType(it_res->residue_name()) == AminoAcid::AAType::unknown) {
103  // We have an heteroatom
104  this->heteroatoms.emplace_back(std::make_shared<Atom>(it_res->residue_name()));
105  total_nb_hetatm++;
106  total_nb_atom++;
107  continue;
108  }
109 
110  // else : It's a normal residue
111  auto current_residue = this->aminoacids.emplace_back(
112  std::shared_ptr<AminoAcid>(new AminoAcid(it_res->residue_name()))
113  );
114 
115  current_residue->setAAId(total_nb_residue);
116 
117  // Iterate over atoms in the residue
118  for (ESBTL::Default_system::Residue::Atoms_const_iterator it_atm = it_res->atoms_begin();
119  it_atm != it_res->atoms_end(); ++it_atm) {
120 
121  assert(it_atm->is_hetatm() == 0); // We expect heteroatoms to have been taken care of previously
122 
123 
124  total_nb_atom++;
125 
126  auto current_atom = current_residue->atoms.emplace_back(
127  std::shared_ptr<Atom>(new Atom(it_atm->atom_name(), true,
128  stringToResType(it_res->residue_name())))
129  );
130 
131  current_atom->setAtomPosition(std::make_tuple(it_atm->x(), it_atm->y(), it_atm->z()));
132 
133 
134  acc_x(it_atm->x());
135  acc_y(it_atm->y());
136  acc_z(it_atm->z());
137 
138 
139 
140  }
141 
142  assignVariantFlagsForResidueAtom(*current_residue, PDBResidueVariantAssignationType::GeneralPurpose);
143 
144  }
145  }
146 
147 
148  // Post processing
149  for(auto pprocessor : postProcessors )
150  {
151  for(auto& residue: aminoacids)
152  {
153  for(auto& atom: residue->atoms)
154  {
155  pprocessor->processAtomFromProtein(*atom,*residue);
156  }
157  }
158  }
159 
160 
161  this->center_x = mean(acc_x);
162  this->center_y = mean(acc_y);
163  this->center_z = mean(acc_z);
164 
165  this->min_x = min(acc_x);
166  this->min_y = min(acc_y);
167  this->min_z = min(acc_z);
168 
169  this->max_x = max(acc_x);
170  this->max_y = max(acc_y);
171  this->max_z = max(acc_z);
172 
173 
174  BOOST_LOG_TRIVIAL(info) << "Loaded protein : " << filename << "\n --> " << total_nb_model << " models, "
175  << total_nb_chain << " chains, " << total_nb_residue << " residues, "
176  << total_nb_atom << " atoms (" << total_nb_hetatm << " heteroatoms)";
177 
178 #ifdef SMOLDOCK_VERBOSE_DEBUG
179  BOOST_LOG_TRIVIAL(debug) << "Protein centered on : ("
180  << this->center_x << ", " << this->center_y << ", " << this->center_z << ") [n=" <<
181  count(acc_x) << "]";
182 #endif
183 
184  }
185  } else {
186  BOOST_LOG_TRIVIAL(error) << "Could not load protein PDB file: " << filename ;
187  return false;
188  }
189 
190 
191  // Post processing steps to add variant flags
192 /*
193  for(auto& residue : this->aminoacids)
194  {
195  //residue->atoms
196  }
197 */
198 
199 
200  return true;
201  }

203  iProtein Protein::getiProtein() const {
204  /*
205  * struct iProtein {
207  double center_x,center_y,center_z;
208  unsigned long size;
209  std::vector<double> x,y,z;
210  std::vector<unsigned char> type;
212  std::map<unsigned int, std::tuple<unsigned long,unsigned long> > AAId_to_AtomPositionInVect;
213  };
214  */
215  iProtein prot;
216 
217  prot.center_x = this->center_x;
218  prot.center_y = this->center_y;
219  prot.center_z = this->center_z;
220 
221 
222  for(auto& residue: this->aminoacids)
223  {
224  unsigned long size_before = prot.x.size();
225  residue->filliProtein(prot, true /* Skip hydrogen */);
226  unsigned long size_after = prot.x.size();
227 
228  // Visual demo :
229  //
230  // index
231  //
232  // size 0
233  //
234  // index 0 1
235  // [A] [A] For residue A : <begin_idx, end_idx> = <0,1> = <size_before, size_after -1>
236  // size 0 > 1 > 2
237  //
238  // index 0 1 2 3 4
239  // [A] [A] [B] [B] [B] For residue B : <begin_idx, end_idx> = <2,4> = <size_before, size_after -1>
240  // size 0 > 1 > 2 > 3 > 4 > 5
241  //
242  // The size_before gives us the index of the first atom that belongs to this residue
243  // The size_after lags behind by one.
244  // This works even if there is only one atom added (then begin_idx = end_idx = size_before = size_after-1)
245  // However there is an edge case : when the residue has no atom, this is incorrect, so :
246  assert(size_after > size_before); // At least one atom added
247 
248  prot.AAId_to_AtomPositionInVect[residue->getAAId()] = std::make_tuple(size_before, size_after - 1);
249  }
250 
251  return prot;
252  }

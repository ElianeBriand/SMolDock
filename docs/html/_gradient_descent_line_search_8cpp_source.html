33  bool GradientDescentLineSearch::optimize(arma::mat startingPoint) {
34 
35  arma::mat& paramsToOptimize = startingPoint;
36  arma::mat gradient(paramVectorDimension, 1, arma::fill::zeros);
37 
38  double score_ = this->scoringFunction->EvaluateWithGradient(paramsToOptimize, gradient);
39 
40  BOOST_LOG_TRIVIAL(debug) << "Initial score : " << score_;
41 
42 
43  double oldscore;
44  double score_progress = 100.0;
45 
46 
47  while (score_progress > 0.001) {
48  if (!hasNonZeroComponent(gradient)) {
49  // Score and all gradient components are zero : cannot decide on a direction,
50  // --> Search failure
51  BOOST_LOG_TRIVIAL(debug)
52  << "Gradient components are zero : cannot use gradient descent for this starting position.";
53  return false;
54  }
55 
56 
57  // We do a linesearch
58  // Aka : we want to know how far in the gradient direction we will go
59 
60  double step = this->differential_epsilon / 10;
61  double previous_step = 0.0;
62  double previous_tentativeNewScore = std::numeric_limits<double>::max();
63 
64  // We go further and further in the direction of the gradient, until we are not bettering the score
65  while (true) {
66 
67  arma::mat tentativeNewParams = paramsToOptimize + (step * gradient);
68 
69  double tentativeNewScore = this->scoringFunction->Evaluate(tentativeNewParams);
70 
71  double initial_tentative_scoreDifference = score_ - tentativeNewScore;
72  double previous_tentative_scoreDifference = previous_tentativeNewScore - tentativeNewScore;
73 
74  // Prepare for next iteration
75  previous_tentativeNewScore = tentativeNewScore;
76 
77  if (initial_tentative_scoreDifference > 0
78  && previous_tentative_scoreDifference > 0) {
79  // We are both better than the initial score, and the previous try for step
80  // We try a somewhat larger step
81  previous_step = step;
82  step = step + (this->differential_epsilon / 10);
83  } else {
84  // We are not better on both count
85  break; // We stop the linesearch
86  }
87  }
88 
89  // Analysing the result of the linesearch
90  if (previous_step == 0.0) {
91  // We have not been able to do any step which improves the score
92  // Most probably, it means we are at a (local) minimum
93  break; // Stopping the gradient search
94  }
95 
96  // Else :
97  // The previous step was an improvement, but this one is not
98  // We use previous_step
99 
100  arma::mat newParams = paramsToOptimize + (previous_step * gradient);
101  paramsToOptimize = newParams;
102 
103  oldscore = score_;
104  score_ = this->scoringFunction->EvaluateWithGradient(paramsToOptimize, gradient);
105  this->iterationNumber++;
106  score_progress = oldscore - score_;
107  BOOST_LOG_TRIVIAL(debug) << "New score: " << score_ << " (progress: " << score_progress << " )";
108  }
109 
110  BOOST_LOG_TRIVIAL(debug) << "Finished gradient descent";
111  this->score = score_;
112  result = paramsToOptimize;
113  return true;
114  }

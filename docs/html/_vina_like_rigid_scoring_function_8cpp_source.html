40  double vina_like_rigid_inter_scoring_func(const iConformer &ligand, const iTransform &transform,
41  const iProtein &protein) {
42 
43  assert(!ligand.x.empty());
44  assert(!protein.x.empty());
45  assert(std::abs(transform.rota.norm() - 1) < 0.01);
46 
47 
48  double score_raw = 0;
49 
50 
51  Eigen::Vector3d ProtCenterPosition = {protein.center_x, protein.center_y, protein.center_z};
52 
53  for (unsigned int idxLig = 0; idxLig < ligand.x.size(); idxLig++) {
54  for (unsigned int idxProt = 0; idxProt < protein.x.size(); idxProt++) {
55 
56 
57  Eigen::Vector3d LigPosition = {ligand.x[idxLig], ligand.y[idxLig], ligand.z[idxLig]};
58  applyRigidTransformInPlace(LigPosition, transform);
59 
60  Eigen::Vector3d ProtPosition = {protein.x[idxProt], protein.y[idxProt], protein.z[idxProt]};
61 
62  Eigen::Vector3d distToCenterVector = LigPosition - ProtCenterPosition;
63 
64  double distanceToProteinCenter = distToCenterVector.norm();
65 
66  if (distanceToProteinCenter > (protein.radius - 1)) {
67  score_raw += std::pow((distanceToProteinCenter - protein.radius), 4) + 10;
68  continue;
69  }
70 
71  Eigen::Vector3d distVect = ProtPosition - LigPosition;
72 
73  double atomicRadiusLig = ligand.atomicRadius[idxLig];
74  double atomicRadiusProt = protein.atomicRadius[idxProt];
75  double rawDist = distVect.norm();
76 
77 
78  const double cutoff = 8.0;
79  if (rawDist >= cutoff)
80  continue;
81 
82  double radToRemove = (atomicRadiusLig + atomicRadiusProt);
83  double distance = rawDist - radToRemove;
84 
85  score_raw += scoreForAtomCouple(distance, ligand.type[idxLig], ligand.variant[idxLig],
86  protein.type[idxProt], protein.variant[idxProt]);
87 /*
88  // Special rubber band term not in Vina scoring func
89  double x_diff_center = std::pow(LigPosition.x - protein.center_x, 2);
90  double y_diff_center = std::pow(LigPosition.y - protein.center_y, 2);
91  double z_diff_center = std::pow(LigPosition.z - protein.center_z, 2);
92  double rawDistCenter = std::sqrt(x_diff_center + y_diff_center + z_diff_center);
93 
94  score += 0.00001 * rawDistCenter;
95 */
96 
97  } // for
98  } // for
99  /*
100  BOOST_LOG_TRIVIAL(debug) << "Intermolecular scoring contribution without weighting :";
101  BOOST_LOG_TRIVIAL(debug) << "Gauss 1 : " << std::fixed<< std::setprecision(5) << score_gauss1;
102  BOOST_LOG_TRIVIAL(debug) << "Gauss 2 : " << std::fixed<< std::setprecision(5) << score_gauss2;
103  BOOST_LOG_TRIVIAL(debug) << "Repulsion : " << std::fixed<< std::setprecision(5) << score_repulsion;
104  BOOST_LOG_TRIVIAL(debug) << "Hydrophobic : " << std::fixed<< std::setprecision(5) << score_hydrophobic;
105  BOOST_LOG_TRIVIAL(debug) << "Hydrogen : " << std::fixed<< std::setprecision(5) << score_hydrogen;
106  BOOST_LOG_TRIVIAL(debug) << "------------------------------------------";
107  BOOST_LOG_TRIVIAL(debug) << "Raw Score : " << score_raw;
108  BOOST_LOG_TRIVIAL(debug) << "Nrotatable : " << ligand.num_rotatable_bond;
109  //*/
110  double final_score = score_raw / (1 + (0.058459999999999998 * ligand.num_rotatable_bond));
111  /*
112  BOOST_LOG_TRIVIAL(debug) << "------------------------------------------";
113  BOOST_LOG_TRIVIAL(debug) << "Final Score : " << final_score;
114  //*/
115 
116  return final_score;
117  }

119  double VinaLikeRigidScoringFunction::Evaluate(const arma::mat &x) {

122  iTransform tr = this->internalToExternalRepr(x);

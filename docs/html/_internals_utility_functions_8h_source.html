103  inline void applyBondRotationInPlace(SmolDock::iConformer &conformer, const iTransform &tr) {
104  for (unsigned int rotBondGroupIdx = 0; rotBondGroupIdx < tr.bondRotationsAngles.size(); rotBondGroupIdx++) {
105  const double &rotAngleForBond = tr.bondRotationsAngles[rotBondGroupIdx];
106  unsigned int atom1Idx = conformer.bondEnds1Index[rotBondGroupIdx];
107  unsigned int atom2Idx = conformer.bondEnds2Index[rotBondGroupIdx];
108 
109 // std::cout << "-- Rotation --\n";
110 // std::cout << " Atom 1 : " << atom1Idx << " " << conformer.x[atom1Idx] << "," << conformer.y[atom1Idx] << "," << conformer.z[atom1Idx] << "\n";
111 // std::cout << " Atom 2 : " << atom2Idx << " "<< conformer.x[atom2Idx] << "," << conformer.y[atom2Idx] << "," << conformer.z[atom2Idx] << "\n";
112 
113  Eigen::Vector3d axis = {
114  (conformer.x[atom2Idx] - conformer.x[atom1Idx]),
115  (conformer.y[atom2Idx] - conformer.y[atom1Idx]),
116  (conformer.z[atom2Idx] - conformer.z[atom1Idx])
117  };
118 
119  Eigen::Translation<double, 3> moveFromOrigin(conformer.x[atom1Idx], conformer.y[atom1Idx],
120  conformer.z[atom1Idx]);
121  Eigen::Translation<double, 3> moveToOrigin = moveFromOrigin.inverse();
122 
123  // std::cout << " Axis : " << axis.x() << "," << axis.y() << "," << axis.z() << " norm = " << axis.norm() << "\n";
124 
125  axis.normalize();
126 
127 // std::cout << " Axis : " << axis.x() << "," << axis.y() << "," << axis.z() << " norm = " << axis.norm() << "\n";
128 
129 
130  assert((axis.norm() - 1) < 0.01);
131 
132  Eigen::AngleAxis<double> rotation(rotAngleForBond, axis);
133 
134  Eigen::Matrix<double, 3, 3> rotMatrix = rotation.toRotationMatrix();
135 
136 // std::cout << "RotMatrix \n" << rotMatrix << "\n";
137 
138  for (unsigned int &rotatedAtomIdx: conformer.rotatableGroups[rotBondGroupIdx]) {
139 // std::cout << " Affecting : " << rotatedAtomIdx << "\n";
140  Eigen::Vector3d startPosition(conformer.x[rotatedAtomIdx],
141  conformer.y[rotatedAtomIdx],
142  conformer.z[rotatedAtomIdx]);
143 
144  Eigen::Vector3d endPosition = moveFromOrigin * (rotMatrix * (moveToOrigin * startPosition));
145 
146  conformer.x[rotatedAtomIdx] = endPosition.x();
147  conformer.y[rotatedAtomIdx] = endPosition.y();
148  conformer.z[rotatedAtomIdx] = endPosition.z();
149  }
150 
151 // std::cout << "-- END Rotation --" << "\n";
152 
153  }
154  }

69  inline double
70  scoreForAtomCouple(double distance, unsigned char atom1AtomicNumber, unsigned int atom1AtomVariant,
71  unsigned char atom2AtomicNumber, unsigned int atom2AtomVariant) {
72 
73  double score_intermol = 0.0;
74 
75  double score_gauss1 = 0;
76  double score_gauss2 = 0;
77  double score_repulsion = 0;
78  double score_hydrophobic = 0;
79  double score_hydrogen = 0;
80 
81  // exp −(d/0.5Å)^2
82  double gauss1 = std::exp(-1 * std::pow(distance / 0.5, 2));
83  score_intermol += -0.035579 * gauss1;
84  score_gauss1 += gauss1;
85 
86  // exp −((d−3Å)/2Å)^2
87  double gauss2 = std::exp(-1 * std::pow((distance - 3.0) / 2.0, 2));
88  score_intermol += -0.005156 * gauss2;
89  score_gauss2 += gauss2;
90 
91  if (distance < 0) {
92  double repuls = std::pow(distance, 2);
93  score_intermol += 0.840245 * repuls;
94  score_repulsion += repuls;
95  }
96 
97  if (isHydrophobic(atom1AtomicNumber, atom1AtomVariant) &&
98  isHydrophobic(atom2AtomicNumber, atom2AtomVariant)) // "Hydrophobic" atoms
99  {
100  double hydrophobic_contrib = 0;
101  if (distance >= 1.5)
102  hydrophobic_contrib = 0;
103  if (distance <= 0.5)
104  hydrophobic_contrib = 1;
105  if (0.5 < distance && distance < 1.5)
106  hydrophobic_contrib = (1.5 - distance);
107 
108  score_intermol += -0.035069 * hydrophobic_contrib;
109  score_hydrophobic += hydrophobic_contrib;
110  }
111 
112  if (hydrogenBondingPossible(atom1AtomicNumber, atom1AtomVariant, atom2AtomicNumber,
113  atom2AtomVariant)) // Hydrogen donor and acceptor
114  {
115  double hbond_contrib = 0;
116  if (distance < -0.7) {
117  hbond_contrib = 1;
118  } else if (distance < 0) // // ==> distance between -0.7 and 0
119  {
120  hbond_contrib = -distance / 0.7;
121  }
122  score_intermol += -0.587439 * hbond_contrib;
123  score_hydrogen += hbond_contrib;
124 
125  }
126  return score_intermol;
127  }

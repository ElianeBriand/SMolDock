132  void ConformerRigidDockingEngine::runDockingEngine() {
133  using namespace boost::accumulators;
134  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_score;
135  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_duration;
136 
137  std::uniform_int_distribution<unsigned int> dis_uint(0, std::numeric_limits<unsigned int>::max());
138 
139 
140  record_timings(begin_docking);
141 
142  for (auto &conformer : this->viConformers) {
143 
144 
145  record_timings(begin_docking_this_conformer);
146 
147 
148 
149 
150  iTransform starting_pos_tr = iTransformIdentityInit();
151  starting_pos_tr.transl = conformer.centroidNormalizingTransform;
152 
153  // Remove this for production
154  // This is to test the local optimizer by shaking up the initial position
155  starting_pos_tr.transl.x += -1.0;
156  starting_pos_tr.transl.y += 1.0;
157  starting_pos_tr.rota.s += 1.0;
158  starting_pos_tr.rota.u += 1.0;
159  starting_pos_tr.rota.v += 1.0;
160  starting_pos_tr.rota.t += 1.0;
161  normalizeQuaternionInPlace(starting_pos_tr.rota);
162 
163  std::unique_ptr<Score::ScoringFunction> vinalike = scoringFunctionFactory(scoringFuncType,
164  conformer,
165  this->protein,
166  starting_pos_tr,
167  1e-3);
168 
169  Optimizer::L_BFGS lbfgs(vinalike.get());
170 
171  Heuristics::RandomRestart heur(vinalike.get(), &lbfgs, dis_uint(this->rnd_generator));
172 
173  heur.search();
174 
175  auto rawResultMatrix = heur.getResultMatrix();
176  double score = vinalike->Evaluate(rawResultMatrix);
177  iConformer result = vinalike->getConformerForParamMatrix(rawResultMatrix);
178 
179  double direct_score = Score::vina_like_rigid_inter_scoring_func(conformer, starting_pos_tr,
180  this->protein);
181  BOOST_LOG_TRIVIAL(debug) << "Score after L-BFGS: " << score << " (from: " << direct_score << ")";
182 
183 
184  this->scores.push_back(score);
185  this->final_iConformer.push_back(result);
186 
187  record_timings(end_docking_this_conformer);
188 
189  acc_score(score);
190  acc_duration(
191  static_cast< std::chrono::duration<double> >(end_docking_this_conformer -
192  begin_docking_this_conformer).count()
193  );
194 
195  }
196 
197  record_timings(end_docking);
198 
199 #ifdef SMOLDOCK_VERBOSE_DEBUG
200  auto total_docking_duration = static_cast< std::chrono::duration<double> >(end_docking -
201  begin_docking).count();
202 
203  BOOST_LOG_TRIVIAL(debug) << "Timings ConformerRigidDockingEngine docking runs"
204  << "\n TOTAL: " << total_docking_duration << "s"
205  << "\n Mean per conformer: " << mean(acc_duration) << "s"
206  << "\n StdDev per conformer: " << std::sqrt(moment<2>(acc_duration)) << "s";
207 #endif
208 
209  BOOST_LOG_TRIVIAL(info) << "Results:";
210  BOOST_LOG_TRIVIAL(info) << " Score Mean: " << mean(acc_score);
211  BOOST_LOG_TRIVIAL(info) << " Score StdDev: " << std::sqrt(moment<2>(acc_score));
212  BOOST_LOG_TRIVIAL(info) << " Scores: ";
213  for (auto &score: scores) {
214  BOOST_LOG_TRIVIAL(info) << " " << score;
215  }
216 
217  }

221  for (const auto &confr : this->final_iConformer) {

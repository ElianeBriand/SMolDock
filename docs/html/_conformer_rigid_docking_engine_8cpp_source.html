84  bool ConformerRigidDockingEngine::setupDockingEngine() {
85 
86  record_timings(begin_setup);
87 
88  std::uniform_int_distribution<> dis_int(0, std::numeric_limits<int>::max());
89  std::uniform_real_distribution<double> dis_real_position(-100.0, 100.0);
90 
91  record_timings(begin_conformersgen);
92 
93  this->orig_ligand->generateConformers(this->viConformers, this->conformer_num, true,
94  dis_int(this->rnd_generator));
95 
96  record_timings(end_conformersgen);
97 
98  if (this->dockBoxSettings.type == DockingBoxSetting::Type::centeredAround) {
99  this->protein = this->orig_protein->getPartialiProtein_sphere(this->dockBoxSettings.center,
100  this->dockBoxSettings.radius, 2.0);
101  } else {
102  this->protein = this->orig_protein->getiProtein();
103  }
104 
105  this->fullProtein = this->orig_protein->getiProtein();
106 
107 
108  record_timings(end_iprotgen);
109 
110 
111  record_timings(end_setup);
112 
113 
114 #ifdef SMOLDOCK_VERBOSE_DEBUG
115  auto total_setup_duration = static_cast< std::chrono::duration<double> >(end_setup - begin_setup).count();
116  auto duration_conformergen = static_cast< std::chrono::duration<double> >(end_conformersgen -
117  begin_conformersgen).count();
118  auto duration_iprot = static_cast< std::chrono::duration<double> >(end_iprotgen -
119  end_conformersgen).count();
120 
121  BOOST_LOG_TRIVIAL(debug) << "Timings ConformerRigidDockingEngine setup"
122  << "\n TOTAL: "
123  << total_setup_duration << "s"
124  << "\n Conformer: "
125  << duration_conformergen
126  << "s [n=" << this->conformer_num << "->" << this->viConformers.size() << "] "
127  << duration_conformergen / this->conformer_num << "s each"
128  << "\n iProt: " << duration_iprot << "s";
129 #endif
130  BOOST_LOG_TRIVIAL(info) << "Conformer docking engine: ready";
131  return true;
132  }

134  void ConformerRigidDockingEngine::runDockingEngine() {
135  using namespace boost::accumulators;
136  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_score;
137  accumulator_set<double, stats<tag::mean, tag::moment<2> > > acc_duration;
138 
139  std::uniform_int_distribution<unsigned int> dis_uint(0, std::numeric_limits<unsigned int>::max());
140 
141 
142  record_timings(begin_docking);
143 
144 
145  for (auto &conformer : this->viConformers) {
146 
147 
148  record_timings(begin_docking_this_conformer);
149 
150 
151  iTransform starting_pos_tr = iTransformIdentityInit(conformer.num_rotatable_bond);
152  starting_pos_tr.transl = conformer.centroidNormalizingTransform;
153 
154 
155  DockingBoxSetting dbsettings = this->dockBoxSettings;
156  unsigned int seed = dis_uint(this->rnd_generator);
157 
158  arma::arma_rng::set_seed(seed);
159 
160  if (dbsettings.type == DockingBoxSetting::Type::centeredAround) {
161  starting_pos_tr.transl.x() += dbsettings.center[0];
162  starting_pos_tr.transl.y() += dbsettings.center[1];
163  starting_pos_tr.transl.z() += dbsettings.center[2];
164  }
165 
166  starting_pos_tr.rota.normalize();
167 
168  std::shared_ptr<Score::ScoringFunction> scoringFunction = scoringFunctionFactory(this->scoringFuncType,
169  conformer,
170  this->protein,
171  starting_pos_tr,
172  1e-3);
173 
174  std::shared_ptr<Optimizer::Optimizer> localOptimizer = optimizerFactory(this->localOptimizerType,
175  scoringFunction.get(),
176  1e-3);
177 
178  Heuristics::HeuristicParameters hParams = Heuristics::heuristicParametersFactory(heuristicType);
179 
180  if (heuristicType == Heuristics::GlobalHeuristicType::IteratedLocalSearch) {
181 
182  if (dbsettings.type == DockingBoxSetting::Type::centeredAround) {
183  std::get<Heuristics::IteratedLocalSearch::Parameters>(
184  hParams).proteinMaxRadius = dbsettings.radius;
185  } else {
186  double protRadius = this->orig_protein->getMaxRadius();
187  std::get<Heuristics::IteratedLocalSearch::Parameters>(hParams).proteinMaxRadius = protRadius;
188  }
189  }
190 
191  if (heuristicType == Heuristics::GlobalHeuristicType::RandomRestart) {
192  if (dbsettings.type == DockingBoxSetting::Type::centeredAround) {
193  std::get<Heuristics::RandomRestart::Parameters>(
194  hParams).proteinMaxRadius = dbsettings.radius;
195  } else {
196  double protRadius = this->orig_protein->getMaxRadius();
197  std::get<Heuristics::RandomRestart::Parameters>(hParams).proteinMaxRadius = protRadius;
198  }
199  }
200 
201 
202  for (unsigned int retryNum = 0; retryNum < this->retryPerConformer; retryNum++) {
203 
204  std::shared_ptr<Heuristics::GlobalHeuristic> globalHeuristic = globalHeuristicFactory(heuristicType,
205  scoringFunction.get(),
206  localOptimizer.get(),
207  seed +
208  retryNum,
209  hParams);
210 
211  globalHeuristic->search();
212 
213  auto rawResultMatrix = globalHeuristic->getResultMatrix();
214  double score = scoringFunction->Evaluate(rawResultMatrix);
215 
216 
217  iConformer result = scoringFunction->getConformerForParamMatrix(rawResultMatrix);
218 
219  double starting_score = Score::vina_like_rigid_inter_scoring_func(conformer, starting_pos_tr,
220  this->protein);
221 
222  double real_score = Score::vina_like_rigid_inter_scoring_func(result, iTransformIdentityInit(),
223  this->fullProtein);
224 
225  BOOST_LOG_TRIVIAL(debug) << "Starting score : " << starting_score;
226  BOOST_LOG_TRIVIAL(debug) << "Score after L-BFGS: " << score;
227  BOOST_LOG_TRIVIAL(debug) << "Real score (full prot): " << real_score;
228 
229  if (score != 0.0) {
230  this->startingScores.push_back(starting_score);
231  this->localScores.push_back(score);
232  this->scores.push_back(real_score);
233  this->allGeneratediConformer.push_back(result);
234  acc_score(real_score);
235  }
236 
237  }
238 
239 
240  record_timings(end_docking_this_conformer);
241 
242 
243 #ifdef SMOLDOCK_VERBOSE_DEBUG
244  acc_duration(
245  static_cast< std::chrono::duration<double> >(end_docking_this_conformer -
246  begin_docking_this_conformer).count()
247  );
248 #endif
249 
250  }
251 
252  record_timings(end_docking);
253 
254 #ifdef SMOLDOCK_VERBOSE_DEBUG
255  auto total_docking_duration = static_cast< std::chrono::duration<double> >(end_docking -
256  begin_docking).count();
257 
258  BOOST_LOG_TRIVIAL(debug) << "Timings ConformerRigidDockingEngine docking runs"
259  << "\n TOTAL: " << total_docking_duration << "s"
260  << "\n Mean per conformer: " << mean(acc_duration) << "s"
261  << "\n StdDev per conformer: " << std::sqrt(moment<2>(acc_duration)) << "s";
262 #endif
263 
264  BOOST_LOG_TRIVIAL(info) << "Results:";
265  BOOST_LOG_TRIVIAL(info) << " Score Mean: " << mean(acc_score);
266  BOOST_LOG_TRIVIAL(info) << " Score StdDev: " << std::sqrt(moment<2>(acc_score));
267  BOOST_LOG_TRIVIAL(info) << " Scores: ";
268  for (unsigned int i = 0; i < this->scores.size(); i++) {
269  BOOST_LOG_TRIVIAL(info) << " " << this->startingScores[i] << " -> " << this->localScores[i]
270  << " -> " << this->scores[i];
271  }
272 
273  BOOST_LOG_TRIVIAL(info) << " Best scores : ";
274 
275  std::vector<std::tuple<int, double>> scoreAndIndices;
276  for (unsigned int i = 0; i < this->scores.size(); i++) {
277  scoreAndIndices.push_back(std::make_tuple(i, this->scores[i]));
278  }
279 
280  std::nth_element(std::begin(scoreAndIndices),
281  std::begin(scoreAndIndices) + conformer_num,
282  std::end(scoreAndIndices),
283  [](const std::tuple<int, double> &a, const std::tuple<int, double> &b) {
284  return std::get<1>(a) < std::get<1>(b);
285  });
286 
287  for (unsigned int i = 0; i < conformer_num; i++) {
288  BOOST_LOG_TRIVIAL(info) << " " << this->startingScores[std::get<0>(scoreAndIndices[i])]
289  << " -> " << this->localScores[std::get<0>(scoreAndIndices[i])]
290  << " -> " << this->scores[std::get<0>(scoreAndIndices[i])];
291  this->bestiConformer.push_back(this->allGeneratediConformer[std::get<0>(scoreAndIndices[i])]);
292  }
293 
294  }

296  std::shared_ptr<DockingResult> ConformerRigidDockingEngine::getDockingResult() {
297  auto ret = std::make_shared<DockingResult>();
298  for (const auto &confr : this->bestiConformer) {
299  Molecule finalLigand = this->orig_ligand->deepcopy();
300  finalLigand.updateAtomPositionsFromiConformer(confr);
301  ret->ligandPoses.emplace_back(finalLigand);
302  }
303  return ret;
304  }

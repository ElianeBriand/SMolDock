44  inline bool isHydrophobic(unsigned char atomicNumber, unsigned int variantFlags) {
45  return (atomicNumber == 6 && (variantFlags & ((unsigned int) Atom::AtomVariant::apolar))) || // C && apolar

47  atomicNumber == 17 || // Cl

87  double score_gauss1 = 0;
88  double score_gauss2 = 0;
89  double score_repulsion = 0;
90  double score_hydrophobic = 0;
91  double score_hydrogen = 0;
92 
93  inline double
94  scoreForAtomCouple(double distance, unsigned char atom1AtomicNumber, unsigned int atom1AtomVariant,
95  unsigned char atom2AtomicNumber, unsigned int atom2AtomVariant) {
96 
97  double score_intermol = 0.0;
98 
99 
100  // exp −(d/0.5Å)^2
101  double gauss1 = std::exp(-1 * std::pow(distance / 0.5, 2));
102  score_intermol += -0.035579 * gauss1;
103  score_gauss1 += gauss1;
104 
105  // exp −((d−3Å)/2Å)^2
106  double gauss2 = std::exp(-1 * std::pow((distance - 3.0) / 2.0, 2));
107  score_intermol += -0.005156 * gauss2;
108  score_gauss2 += gauss2;
109 
110  if (distance < 0) {
111  double repuls = std::pow(distance, 2);
112  score_intermol += 0.840245 * repuls;
113  score_repulsion += repuls;
114  }
115 
116  if (isHydrophobic(atom1AtomicNumber, atom1AtomVariant) &&
117  isHydrophobic(atom2AtomicNumber, atom2AtomVariant)) // "Hydrophobic" atoms
118  {
119  double hydrophobic_contrib = 0;
120  if (distance >= 1.5)
121  hydrophobic_contrib = 0;
122  if (distance <= 0.5)
123  hydrophobic_contrib = 1;
124  if (0.5 < distance && distance < 1.5)
125  hydrophobic_contrib = (1.5 - distance);
126 
127  score_intermol += -0.035069 * hydrophobic_contrib;
128  score_hydrophobic += hydrophobic_contrib;
129  }
130 
131  if (hydrogenBondingPossible(atom1AtomicNumber, atom1AtomVariant, atom2AtomicNumber,
132  atom2AtomVariant)) // Hydrogen donor and acceptor
133  {
134  double hbond_contrib = 0;
135  if (distance < -0.7) {
136  hbond_contrib = 1;
137  } else if (distance < 0) // // ==> distance between -0.7 and 0
138  {
139  hbond_contrib = -distance / 0.7;
140  }
141  score_intermol += -0.587439 * hbond_contrib;
142  score_hydrogen += hbond_contrib;
143 
144  }
145  return score_intermol;
146  }

149  double vina_like_rigid_inter_scoring_func(const iConformer &ligand, const iTransform &transform,
150  const iProtein &protein) {
151 
152  assert(ligand.x.size() != 0);
153  assert(protein.x.size() != 0);
154  assert(std::abs(quaternionNorm(transform.rota) - 1) < 0.01);
155 
156 
157  double score_raw = 0;
158 
159 
160  score_gauss1 = 0;
161  score_gauss2 = 0;
162  score_repulsion = 0;
163  score_hydrophobic = 0;
164  score_hydrogen = 0;
165 
166  for (unsigned int idxLig = 0; idxLig < ligand.x.size(); idxLig++) {
167  for (unsigned int idxProt = 0; idxProt < protein.x.size(); idxProt++) {
168 
169 
170  iVect LigPosition = {ligand.x[idxLig], ligand.y[idxLig], ligand.z[idxLig]};
171  applyTransformInPlace(LigPosition, transform);
172 
173 
174  double x_diff_sq = std::pow(LigPosition.x - protein.x[idxProt], 2);
175  double y_diff_sq = std::pow(LigPosition.y - protein.y[idxProt], 2);
176  double z_diff_sq = std::pow(LigPosition.z - protein.z[idxProt], 2);
177 
178 
179  double atomicRadiusLig = ligand.atomicRadius[idxLig];
180  double atomicRadiusProt = protein.atomicRadius[idxProt];
181  double rawDist = std::sqrt(x_diff_sq + y_diff_sq + z_diff_sq);
182 
183 
184  const double cutoff = 8.0;
185  if (rawDist >= cutoff)
186  continue;
187 
188  double radToRemove = (atomicRadiusLig + atomicRadiusProt);
189  double distance = rawDist - radToRemove;
190 
191  score_raw += scoreForAtomCouple(distance, ligand.type[idxLig], ligand.variant[idxLig],
192  protein.type[idxProt], protein.variant[idxProt]);
193 /*
194  // Special rubber band term not in Vina scoring func
195  double x_diff_center = std::pow(LigPosition.x - protein.center_x, 2);
196  double y_diff_center = std::pow(LigPosition.y - protein.center_y, 2);
197  double z_diff_center = std::pow(LigPosition.z - protein.center_z, 2);
198  double rawDistCenter = std::sqrt(x_diff_center + y_diff_center + z_diff_center);
199 
200  score += 0.00001 * rawDistCenter;
201 */
202 
203  } // for
204  } // for
205  /*
206  BOOST_LOG_TRIVIAL(debug) << "Intermolecular scoring contribution without weighting :";
207  BOOST_LOG_TRIVIAL(debug) << "Gauss 1 : " << std::fixed<< std::setprecision(5) << score_gauss1;
208  BOOST_LOG_TRIVIAL(debug) << "Gauss 2 : " << std::fixed<< std::setprecision(5) << score_gauss2;
209  BOOST_LOG_TRIVIAL(debug) << "Repulsion : " << std::fixed<< std::setprecision(5) << score_repulsion;
210  BOOST_LOG_TRIVIAL(debug) << "Hydrophobic : " << std::fixed<< std::setprecision(5) << score_hydrophobic;
211  BOOST_LOG_TRIVIAL(debug) << "Hydrogen : " << std::fixed<< std::setprecision(5) << score_hydrogen;
212  BOOST_LOG_TRIVIAL(debug) << "------------------------------------------";
213  BOOST_LOG_TRIVIAL(debug) << "Raw Score : " << score_raw;
214  BOOST_LOG_TRIVIAL(debug) << "Nrotatable : " << ligand.num_rotatable_bond;
215  //*/
216  double final_score = score_raw / (1 + (0.058459999999999998 * ligand.num_rotatable_bond));
217  /*
218  BOOST_LOG_TRIVIAL(debug) << "------------------------------------------";
219  BOOST_LOG_TRIVIAL(debug) << "Final Score : " << final_score;
220  //*/
221 
222  return final_score;
223  }

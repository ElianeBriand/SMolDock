34  void assignVariantFlagsForResidueAtom(AminoAcid &residue, PDBResidueVariantAssignationType assignation_type) {
35 
36  std::set<std::tuple<AminoAcid::AAType, std::vector<std::tuple<std::string, Atom::AtomVariant> > > > *ResidueAtomPropertiesLookupTable;
37  if (assignation_type == PDBResidueVariantAssignationType::GeneralPurpose) {
38  ResidueAtomPropertiesLookupTable = &ResidueAtomPropertiesLookupTable_General;
39  } else {
40  ResidueAtomPropertiesLookupTable = &ResidueAtomPropertiesLookupTable_General;
41  }
42 
43  AminoAcid::AAType type = residue.getType();
44  if (type == AminoAcid::AAType::unknown) {
45  BOOST_LOG_TRIVIAL(error)
46  << "Encountered unknown residue type while assigning atom variant. Check PDB file. Undefined behaviour may occur.";
47 
48  } else {
49  auto record_it = std::find_if(std::begin(*ResidueAtomPropertiesLookupTable),
50  std::end(*ResidueAtomPropertiesLookupTable),
51  [&](const std::tuple<AminoAcid::AAType, std::vector<std::tuple<std::string, Atom::AtomVariant> > > &e) {
52  return std::get<0>(e) == type;
53  });
54  if (record_it == std::end(*ResidueAtomPropertiesLookupTable)) {
55  // Not found
56  BOOST_LOG_TRIVIAL(error)
57  << "Amino acid not found in ResidueAtomPropertiesLookupTable. Variant assignation not done for this amino acid.";
58  return;
59  }
60 
61 
62  std::vector<std::tuple<std::string, Atom::AtomVariant> > record_for_this_AA = std::get<1>(*record_it);
63 
64 
65  for (auto &atom : residue.atoms) {
66  std::string complete_atom_name = atom->atomClassInResidue;
67  auto nameTypeTuple_it = std::find_if(std::begin(record_for_this_AA), std::end(record_for_this_AA),
68  [&](const std::tuple<std::string, Atom::AtomVariant> &e) {
69  return std::get<0>(e) == complete_atom_name;
70  });
71  if (nameTypeTuple_it != std::end(record_for_this_AA)) {
72  // We have an association between complete atom name and AtomVariant
73  Atom::AtomVariant newVariant = atom->getAtomVariant() | std::get<1>(*nameTypeTuple_it);
74  atom->setAtomVariant(newVariant);
75  } else {
76  BOOST_LOG_TRIVIAL(error)
77  << "Missing AtomVariant record for residue " << resTypeToString(type) << ", atom "
78  << complete_atom_name;
79  }
80 
81  }
82 
83  }

87  void assignApolarCarbonFlag(std::vector<std::shared_ptr<Atom> > &atomVect) {
88  for (auto &atom: atomVect) {
89  if (atom->getAtomType() == Atom::AtomType::carbon) {
90  bool isLinkedToHeteroatom = false;
91  for (const auto &bond_weakptr: atom->bonds) {
92  std::shared_ptr<Bond> bond = bond_weakptr.lock();
93  if (bond == nullptr) {
94  BOOST_LOG_TRIVIAL(error)
95  << "Encountered nullptr weak_ptr while traversing molecule. This is definitely a bug: please report.";
96  BOOST_LOG_TRIVIAL(error) << "Exiting unsucessfully...";
97  std::exit(6);
98  }
99  std::shared_ptr<Atom> endA = bond->getEndA();
100  std::shared_ptr<Atom> endB = bond->getEndB();
101 
102  assert((endA.get() == atom.get()) !=
103  (endB.get() ==
104  atom.get())); // Check that the current atom is exactly one of the end (!= is logical XOR for bool)
105 
106  std::shared_ptr<Atom> opposite_end;
107  if (endA.get() == atom.get())
108  // The current atom is EndA
109  opposite_end = endB;
110  else if (endB.get() == atom.get())
111  // The current atom is EndB
112  opposite_end = endA;
113 
114  Atom::AtomType linked_atom_type = opposite_end->getAtomType();
115  if ((linked_atom_type != Atom::AtomType::carbon) &&
116  (linked_atom_type != Atom::AtomType::hydrogen)) {
117  isLinkedToHeteroatom = true;
118  }
119  }
120  if (!isLinkedToHeteroatom) {
121  atom->variant = atom->variant | Atom::AtomVariant::apolar;
122  }
123  }
124  }
125  }

33  void MoleculeTraversal::traverse(shared_ptr<Atom> a) {
34  cout << a->getAtomID() << " : " << a->getTypeAsString() << " (" << a->bonds.size() << " bonds)" << endl;
35  for (const auto &bond: a->bonds) {
36  auto b = bond.lock();
37  if (b == nullptr) {
38  cout << "MoleculeTraversal::traverse : Cant lock weak ptr (?)" << endl;
39  }
40  auto endA = b->getEndA();
41  auto endB = b->getEndB();
42  auto nonself_end = (endA == a) ? endB : endA; // Which end of the bond is not this atom
43 
44  for (const auto &already_visited_atom: already_visited_atoms) {
45  if (nonself_end == already_visited_atom) // already visited
46  {
47  goto nextbond; // end traversal, go to next iteration
48  }
49  }
50  already_visited_atoms.push_back(a);
51  traverse(nonself_end); // Continue deeper
52  nextbond:;
53  }
54  }

226  bool Molecule::populateInternalAtomAndBondFromRWMol(unsigned int seed,
227  std::vector<std::shared_ptr<InputPostProcessor::InputPostProcessor> > postProcessors) {
228 
229  // We care about Hs only during conformer generation
230  // TODO: find out if we need to care at other moments...
231  RDKit::MolOps::addHs(*(this->rwmol));
232 
233  this->initial_conformer_id = 0;
234  if (this->rwmol->getNumConformers() == 0) // We need to generate the first conformer
235  {
236  this->initial_conformer_id = RDKit::DGeomHelpers::EmbedMolecule(*rwmol, 1000, seed, true);
237  RDKit::MMFF::MMFFOptimizeMolecule(*rwmol, 1000, "MMFF94s", 10.0, this->initial_conformer_id);
238  } else { // Else we will just use the first
239  this->initial_conformer_id = (*(this->rwmol->beginConformers()))->getId();
240  }
241 
242  RDKit::Conformer &starting_conformer = rwmol->getConformer(this->initial_conformer_id);
243 
244  // We remove them afterward
245  RDKit::MolOps::removeHs((*(this->rwmol)));
246 
247  if (!this->rwmol->hasProp("_Name")) {
248  this->rwmol->setProp("_Name", "ligand");
249  }
250 
251  unsigned int currentAtomIndexInVector = 0;
252  for (auto atom_it = rwmol->beginAtoms(); atom_it != rwmol->endAtoms(); ++atom_it) {
253  std::shared_ptr<Atom> current_atom;
254  // TODO : refactor this using the std::set of type-name-properties
255 
256  Atom::AtomType atomicNumToFind = static_cast<Atom::AtomType>((*atom_it)->getAtomicNum());
257 
258  auto it = std::find_if(Atom::AtomTypeLabel.begin(), Atom::AtomTypeLabel.end(),
259  [atomicNumToFind](
260  const std::tuple<Atom::AtomType, std::string, std::string, double> &e) {
261  return std::get<0>(e) == atomicNumToFind;
262  });
263 
264  if (it == Atom::AtomTypeLabel.end()) {
265  BOOST_LOG_TRIVIAL(error) << "Unsupported atom type";
266  BOOST_LOG_TRIVIAL(error) << "Atomic num = " << (*atom_it)->getAtomicNum();
267  return false;
268  }
269 
270  current_atom = std::make_shared<Atom>(std::get<0>(*it), (*atom_it)->getIdx());
271 
272  const RDGeom::Point3D &position = starting_conformer.getAtomPos((*atom_it)->getIdx());
273  current_atom->setAtomPosition(std::make_tuple(position.x, position.y, position.z));
274 
275  atoms.push_back(current_atom);
276  this->RDKitAtomIdxToAtomsPosInVector[(*atom_it)->getIdx()] = currentAtomIndexInVector;
277  this->atomsPosInVectorToRDKitAtomIdx[currentAtomIndexInVector] = (*atom_it)->getIdx();
278  currentAtomIndexInVector++;
279  }
280 
281  for (auto bond_it = rwmol->beginBonds(); bond_it != rwmol->endBonds(); ++bond_it) {
282  auto beginAtom = (*bond_it)->getBeginAtom();
283  auto endAtom = (*bond_it)->getEndAtom();
284 
285  unsigned int beginAtomID = beginAtom->getIdx();
286  unsigned int endAtomID = endAtom->getIdx();
287 
288  auto resultBegin = std::find_if(std::begin(atoms), std::end(atoms),
289  [beginAtomID](const std::shared_ptr<Atom> &atom) -> bool {
290  return atom->getAtomID() == beginAtomID;
291  }
292  );
293 
294  auto resultEnd = std::find_if(std::begin(atoms), std::end(atoms),
295  [endAtomID](const std::shared_ptr<Atom> &atom) -> bool {
296  return atom->getAtomID() == endAtomID;
297  }
298  );
299 
300  if (resultBegin == std::end(atoms) || resultEnd == std::end(atoms)) {
301  BOOST_LOG_TRIVIAL(error) << "Unable to find atom with ID = " << beginAtomID << " or " << endAtomID;
302  return false;
303  }
304 
305  auto new_bond = std::make_shared<Bond>(*resultBegin, *resultEnd);
306 
307 
308  /*
309  For reference, from RDKit's bond.h. We do not intent to support the exotic/dative bond for the
310  foreseeable future.
311 
312  typedef enum {
313  UNSPECIFIED = 0,
314  SINGLE,
315  DOUBLE,
316  TRIPLE,
317  QUADRUPLE,
318  QUINTUPLE,
319  HEXTUPLE,
320  ONEANDAHALF,
321  TWOANDAHALF,
322  THREEANDAHALF,
323  FOURANDAHALF,
324  FIVEANDAHALF,
325  AROMATIC,
326  IONIC,
327  HYDROGEN,
328  THREECENTER,
329  DATIVEONE, //!< one-electron dative (e.g. from a C in a Cp ring to a metal)
330  DATIVE, //!< standard two-electron dative
331  DATIVEL, //!< standard two-electron dative
332  DATIVER, //!< standard two-electron dative
333  OTHER,
334  ZERO //!< Zero-order bond (from
335  // http://pubs.acs.org/doi/abs/10.1021/ci200488k)
336  } BondType;
337 
338  */
339 
340  switch ((*bond_it)->getBondType()) {
341  case RDKit::Bond::SINGLE:
342  new_bond->setBondType(Bond::BondType::singlebond);
343  break;
344  case RDKit::Bond::DOUBLE:
345  new_bond->setBondType(Bond::BondType::doublebond);
346  break;
347  case RDKit::Bond::TRIPLE:
348  new_bond->setBondType(Bond::BondType::triplebond);
349  break;
350  case RDKit::Bond::AROMATIC: // As expected aromatic is set for all bond in the cycle
351  // , not just the kekule-style "double bond"
352  new_bond->setBondType(Bond::BondType::aromatic);
353  break;
354  default:
355  BOOST_LOG_TRIVIAL(error) << "Unsupported bond type";
356  BOOST_LOG_TRIVIAL(error) << "Enum bond type = " << (*bond_it)->getBondType();
357  return false;
358  // break;
359  }
360  new_bond->publicizeToAtom();
361  bonds.push_back(new_bond);
362  }
363 
364  this->numberOfRotatableBonds = RDKit::Descriptors::calcNumRotatableBonds((RDKit::ROMol) (*this->rwmol));
365 
366 
367  unsigned int numAliphaticRing = RDKit::Descriptors::calcNumAliphaticRings((RDKit::ROMol) (*this->rwmol));
368  if (numAliphaticRing > 0) {
369  BOOST_LOG_TRIVIAL(error) << "Non-rigid (ie, non-aromatic) rings are unsupported yet";
370  BOOST_LOG_TRIVIAL(error) << " Found " << numAliphaticRing << " aliphatic ring(s)";
371  BOOST_LOG_TRIVIAL(error) << " Processing will continue but correctness is not guaranteed";
372  }
373 
374 
375 
376 
377  // rotatable bond SMARTS from RdKit source (Lipinski.cpp)
378  std::string strict_pattern =
379  "[!$(*#*)&!D1&!$(C(F)(F)F)&!$(C(Cl)(Cl)Cl)&!$(C(Br)(Br)Br)&!$(C([CH3])("
380  "[CH3])[CH3])&!$([CD3](=[N,O,S])-!@[#7,O,S!D1])&!$([#7,O,S!D1]-!@[CD3]="
381  "[N,O,S])&!$([CD3](=[N+])-!@[#7!D1])&!$([#7!D1]-!@[CD3]=[N+])]-!@[!$(*#"
382  "*)&!D1&!$(C(F)(F)F)&!$(C(Cl)(Cl)Cl)&!$(C(Br)(Br)Br)&!$(C([CH3])([CH3])"
383  "[CH3])]";
384 
385  std::shared_ptr<RDKit::RWMol> rotatableBondPatt(RDKit::SmartsToMol(strict_pattern));
386 
387  std::vector<RDKit::MatchVectType> matchsRotBonds;
388 
389  if (RDKit::SubstructMatch(static_cast<const RDKit::ROMol &>(*this->rwmol), *rotatableBondPatt,
390  matchsRotBonds)) {
391  BOOST_LOG_TRIVIAL(debug) << "Rotatable bond(s) found";
392  }
393 
394  assert(matchsRotBonds.size() == this->numberOfRotatableBonds);
395 
396  for (unsigned int i = 0; i < matchsRotBonds.size(); ++i) {
397  if (matchsRotBonds[i].size() < 2) {
398  BOOST_LOG_TRIVIAL(error) << "Rotatable bond pattern matching yielded 1 end only ??";
399  continue;
400  }
401  unsigned int atom1Index = matchsRotBonds[i][0].second;
402  unsigned int atom2Index = matchsRotBonds[i][1].second;
403  const RDKit::Atom *atomMatched1 = this->rwmol->getAtomWithIdx(atom1Index);
404  const RDKit::Atom *atomMatched2 = this->rwmol->getAtomWithIdx(atom2Index);
405 
406  assert(atomMatched1 != nullptr);
407  assert(atomMatched2 != nullptr);
408 
409  BOOST_LOG_TRIVIAL(debug) << "Rotatable bond found : " << atomMatched1->getSymbol()
410  << "-" << atomMatched2->getSymbol() << "(" << atomMatched1->getIdx() << "-"
411  << atomMatched2->getIdx() << ")";
412 
413  auto itFoundBond = std::find_if(std::begin(this->bonds), std::end(this->bonds),
414  [atom1Index, atom2Index](const std::shared_ptr<Bond> &e) {
415  // We might get endA and endB in either order
416  return ((e->getEndA()->getAtomID() == atom1Index) &&
417  (e->getEndB()->getAtomID() == atom2Index))
418  || ((e->getEndB()->getAtomID() == atom1Index) &&
419  (e->getEndA()->getAtomID() == atom2Index));
420  });
421  if (itFoundBond == std::end(this->bonds)) {
422  BOOST_LOG_TRIVIAL(error)
423  << "Rotatable bond not found in Molecule::bonds (but present in RDKit::RWMol) ?";
424  return false;
425  }
426 
427  std::vector<unsigned int> atomIdxOfRotatedAtoms;
428 
429 
430  // TODO: while either side of the bond can be designated as "the part that rotate"
431  // it would be useful to designate the smallest one, so as to do less math
432  // or do some kind of tree traversal to rationalize all this "subsection to rotate" business
433  // (this will help very much for quaternion, as they compose)
434 
435  std::function<void(const RDKit::Atom *, const RDKit::Atom *, const RDKit::Atom *)> exploreAdj;
436  exploreAdj = ([&atomIdxOfRotatedAtoms, this, &exploreAdj]
437  (const RDKit::Atom *atom,
438  const RDKit::Atom *avoidThisAtom,
439  const RDKit::Atom *avoidThisOtherAtom) {
440  RDKit::ROMol::ADJ_ITER nbr, end_nbr;
441  boost::tie(nbr, end_nbr) = this->rwmol->getAtomNeighbors(atom);
442  while (nbr != end_nbr) {
443  const RDKit::Atom *nbr_atom = (*this->rwmol)[*nbr];
444  if (nbr_atom == avoidThisAtom ||
445  nbr_atom == avoidThisOtherAtom ||
446  nbr_atom == atom ||
447  std::find(atomIdxOfRotatedAtoms.begin(), atomIdxOfRotatedAtoms.end(), nbr_atom->getIdx()) !=
448  atomIdxOfRotatedAtoms.end()) {
449  ++nbr;
450  continue;
451  }
452  atomIdxOfRotatedAtoms.push_back(nbr_atom->getIdx());
453  //BOOST_LOG_TRIVIAL(debug) << "AdjAtom from " << atom->getIdx() <<"(" <<atom->getSymbol() << ")" <<
454  //" : " << nbr_atom->getIdx() << " : " << nbr_atom->getSymbol();
455  exploreAdj(nbr_atom, avoidThisAtom, avoidThisOtherAtom);
456  ++nbr;
457  }
458  });
459 
460  exploreAdj(atomMatched1, atomMatched1, atomMatched2);
461 
462  this->rotatableBonds.emplace_back(
463  std::make_tuple(*itFoundBond, atom1Index, atom2Index, atomIdxOfRotatedAtoms));
464 
465 
466  }
467 
468 
469 
470 
471 
472  // Post processing to assign variant
473 
474  // Processing to assign the apolar flag to carbon
475  assignApolarCarbonFlag(this->atoms);
476 
477  // Final post processing : calling
478  for (const auto &pprocessor : postProcessors) {
479  for (auto &atom: this->atoms) {
480  pprocessor->processAtomFromLigand(*atom);
481  }
482  }
483 
484  return true;
485  }

502  Molecule Molecule::deepcopy() {
503  Molecule newmol(*this);
504 
505 
506  std::vector<std::tuple<std::shared_ptr<Atom>, Atom *> > new_old_lookup_table;
507  newmol.atoms.clear();
508  for (auto &ptr: this->atoms) {
509  Atom *newatom_ptr = new Atom(*ptr);
510  newatom_ptr->bonds.clear(); // We will restore it after rebuilding the bonds
511  std::shared_ptr<Atom> &inserted = newmol.atoms.emplace_back(newatom_ptr);
512  new_old_lookup_table.push_back(
513  std::make_tuple(inserted, ptr.get())); // Register new/old pair to rebuild bonds
514 
515  }
516 
517  newmol.bonds.clear();
518  for (auto &ptr: this->bonds) {
519  Atom *endA_ptr = ptr->getEndA().get();
520  Atom *endB_ptr = ptr->getEndB().get();
521 
522  // Find matches in lookup table :
523  std::shared_ptr<Atom> endA = std::get<0>(
524  *(std::find_if(std::begin(new_old_lookup_table), std::end(new_old_lookup_table),
525  [&](const std::tuple<std::shared_ptr<Atom>, Atom *> &e) {
526  return (std::get<1>(e) == endA_ptr);
527  })));
528 
529  std::shared_ptr<Atom> endB = std::get<0>(
530  *(std::find_if(std::begin(new_old_lookup_table), std::end(new_old_lookup_table),
531  [&](const std::tuple<std::shared_ptr<Atom>, Atom *> &e) {
532  return (std::get<1>(e) == endB_ptr);
533  })));
534 
535  Bond *newbond = new Bond(endA, endB, ptr->getBondID());
536  newbond->setBondType(ptr->getBondType());
537  std::shared_ptr<Bond> &inserted_bond = newmol.bonds.emplace_back(newbond);
538  // PublicizeToAtom uses enable_shared_from_this, which means the previous statement, creating a shared_ptr
539  // must occur before the following statement, lest throw std::bad_weak_ptr
540  inserted_bond->publicizeToAtom();
541  }
542 
543  newmol.rwmol.reset(new RDKit::RWMol(*rwmol));
544 
545  return newmol;
546  }

555  iConformer Molecule::generateIConformerForGivenRDKitConformerID(unsigned int id,
556  bool centroidNormalization) const {
557  iConformer conformer;
558  RDKit::Conformer rdkit_conformer = this->rwmol->getConformer(id);
559 
560  unsigned long num_atoms = this->atoms.size();
561  conformer.x.insert(conformer.x.begin(), num_atoms, 0.0);
562  conformer.y.insert(conformer.y.begin(), num_atoms, 0.0);
563  conformer.z.insert(conformer.z.begin(), num_atoms, 0.0);
564  conformer.variant.insert(conformer.variant.begin(), num_atoms, 0);
565  conformer.type.insert(conformer.type.begin(), num_atoms, 0);
566  conformer.atomicRadius.insert(conformer.atomicRadius.begin(), num_atoms, 0.0);
567 
568  conformer.num_rotatable_bond = this->numberOfRotatableBonds;
569 
570  conformer.bondEnds1Index.insert(conformer.bondEnds1Index.begin(),
571  conformer.num_rotatable_bond,
572  0);
573  conformer.bondEnds2Index.insert(conformer.bondEnds2Index.begin(),

716

52  std::vector< std::shared_ptr<InputPostProcessor::InputPostProcessor> > postProcessors) {

169  bool Molecule::populateInternalAtomAndBondFromRWMol(unsigned int seed,
170  std::vector< std::shared_ptr<InputPostProcessor::InputPostProcessor> > postProcessors) {
171 
172  // We care about Hs only during conformer generation
173  // TODO: find out if we need to care at other moments...
174  RDKit::MolOps::addHs(*(this->rwmol));
175 
176  this->initial_conformer_id = 0;
177  if (this->rwmol->getNumConformers() == 0) // We need to generate the first conformer
178  {
179  this->initial_conformer_id = RDKit::DGeomHelpers::EmbedMolecule(*rwmol, 1000, seed, true);
180  } else { // Else we will just use the first
181  this->initial_conformer_id = (*(this->rwmol->beginConformers()))->getId();
182  }
183 
184  RDKit::Conformer &starting_conformer = rwmol->getConformer(this->initial_conformer_id);
185 
186  // We remove them afterward
187  RDKit::MolOps::removeHs((*(this->rwmol)));
188 
189 
190  for (auto atom_it = rwmol->beginAtoms(); atom_it != rwmol->endAtoms(); ++atom_it) {
191  std::shared_ptr<Atom> current_atom;
192  // TODO : refactor this using the std::set of type-name-properties
193  switch ((*atom_it)->getAtomicNum()) {
194  case 1:
195  current_atom = std::make_shared<Atom>(Atom::AtomType::hydrogen, (*atom_it)->getIdx());
196  break;
197  case 6:
198  current_atom = std::make_shared<Atom>(Atom::AtomType::carbon, (*atom_it)->getIdx());
199  break;
200  case 7:
201  current_atom = std::make_shared<Atom>(Atom::AtomType::nitrogen, (*atom_it)->getIdx());
202  break;
203  case 8:
204  current_atom = std::make_shared<Atom>(Atom::AtomType::oxygen, (*atom_it)->getIdx());
205  break;
206  default:
207  BOOST_LOG_TRIVIAL(error) << "Unsupported atom type";
208  BOOST_LOG_TRIVIAL(error) << "Atomic num = " << (*atom_it)->getAtomicNum();
209  return false;
210  // break;
211  }
212  const RDGeom::Point3D &position = starting_conformer.getAtomPos((*atom_it)->getIdx());
213  current_atom->setAtomPosition(std::make_tuple(position.x, position.y, position.z));
214 
215 
216  atoms.push_back(current_atom);
217  }
218 
219  for (auto bond_it = rwmol->beginBonds(); bond_it != rwmol->endBonds(); ++bond_it) {
220  auto beginAtom = (*bond_it)->getBeginAtom();
221  auto endAtom = (*bond_it)->getEndAtom();
222 
223  unsigned int beginAtomID = beginAtom->getIdx();
224  unsigned int endAtomID = endAtom->getIdx();
225 
226  auto resultBegin = std::find_if(std::begin(atoms), std::end(atoms),
227  [beginAtomID](const std::shared_ptr<Atom> &atom) -> bool {
228  return atom->getAtomID() == beginAtomID;
229  }
230  );
231 
232  auto resultEnd = std::find_if(std::begin(atoms), std::end(atoms),
233  [endAtomID](const std::shared_ptr<Atom> &atom) -> bool {
234  return atom->getAtomID() == endAtomID;
235  }
236  );
237 
238  if (resultBegin == std::end(atoms) || resultEnd == std::end(atoms)) {
239  BOOST_LOG_TRIVIAL(error) << "Unable to find atom with ID = " << beginAtomID << " or " << endAtomID;
240  return false;
241  }
242 
243  auto new_bond = std::make_shared<Bond>(*resultBegin, *resultEnd);
244 
245 
246  /*
247  For reference, from RDKit's bond.h. We do not intent to support the exotic/dative bond for the
248  foreseeable future.
249 
250  typedef enum {
251  UNSPECIFIED = 0,
252  SINGLE,
253  DOUBLE,
254  TRIPLE,
255  QUADRUPLE,
256  QUINTUPLE,
257  HEXTUPLE,
258  ONEANDAHALF,
259  TWOANDAHALF,
260  THREEANDAHALF,
261  FOURANDAHALF,
262  FIVEANDAHALF,
263  AROMATIC,
264  IONIC,
265  HYDROGEN,
266  THREECENTER,
267  DATIVEONE, //!< one-electron dative (e.g. from a C in a Cp ring to a metal)
268  DATIVE, //!< standard two-electron dative
269  DATIVEL, //!< standard two-electron dative
270  DATIVER, //!< standard two-electron dative
271  OTHER,
272  ZERO //!< Zero-order bond (from
273  // http://pubs.acs.org/doi/abs/10.1021/ci200488k)
274  } BondType;
275 
276  */
277 
278  switch ((*bond_it)->getBondType()) {
279  case RDKit::Bond::SINGLE:
280  new_bond->setBondType(Bond::BondType::singlebond);
281  break;
282  case RDKit::Bond::DOUBLE:
283  new_bond->setBondType(Bond::BondType::doublebond);
284  break;
285  case RDKit::Bond::TRIPLE:
286  new_bond->setBondType(Bond::BondType::triplebond);
287  break;
288  case RDKit::Bond::AROMATIC: // As expected aromatic is set for all bond in the cycle
289  // , not just the kekule-style "double bond"
290  new_bond->setBondType(Bond::BondType::aromatic);
291  break;
292  default:
293  BOOST_LOG_TRIVIAL(error) << "Unsupported bond type";
294  BOOST_LOG_TRIVIAL(error) << "Enum bond type = " << (*bond_it)->getBondType();
295  return false;
296  // break;
297  }
298  new_bond->publicizeToAtom();
299  bonds.push_back(new_bond);
300  }
301 
302  this->numberOfRotatableBonds = RDKit::Descriptors::calcNumRotatableBonds((RDKit::ROMol) (*this->rwmol));
303 
304 
305  // Post processing to assign variant
306 
307  // Processing to assign the apolar flag to carbon
308  assignApolarCarbonFlag(this->atoms);
309 
310  // Final post processing : calling
311  for (auto pprocessor : postProcessors) {
312  for (auto &atom: this->atoms) {
313  pprocessor->processAtomFromLigand(*atom);
314  }
315  }
316 
317  return true;
318  }

332  Molecule Molecule::deepcopy() {
333  Molecule newmol(*this);
334 
335 
336  std::vector<std::tuple<std::shared_ptr<Atom>, Atom *> > new_old_lookup_table;
337  newmol.atoms.clear();
338  for (auto &ptr: this->atoms) {
339  Atom *newatom_ptr = new Atom(*ptr);
340  newatom_ptr->bonds.clear(); // We will restore it after rebuilding the bonds
341  std::shared_ptr<Atom> &inserted = newmol.atoms.emplace_back(newatom_ptr);
342  new_old_lookup_table.push_back(
343  std::make_tuple(inserted, ptr.get())); // Register new/old pair to rebuild bonds
344 
345  }
346 
347  newmol.bonds.clear();
348  for (auto &ptr: this->bonds) {
349  Atom *endA_ptr = ptr->getEndA().get();
350  Atom *endB_ptr = ptr->getEndB().get();
351 
352  // Find matches in lookup table :
353  std::shared_ptr<Atom> endA = std::get<0>(
354  *(std::find_if(std::begin(new_old_lookup_table), std::end(new_old_lookup_table),
355  [&](const std::tuple<std::shared_ptr<Atom>, Atom *> &e) {
356  return (std::get<1>(e) == endA_ptr);
357  })));
358 
359  std::shared_ptr<Atom> endB = std::get<0>(
360  *(std::find_if(std::begin(new_old_lookup_table), std::end(new_old_lookup_table),
361  [&](const std::tuple<std::shared_ptr<Atom>, Atom *> &e) {
362  return (std::get<1>(e) == endB_ptr);
363  })));
364 
365  Bond *newbond = new Bond(endA, endB, ptr->getBondID());
366  newbond->setBondType(ptr->getBondType());
367  std::shared_ptr<Bond> &inserted_bond = newmol.bonds.emplace_back(newbond);
368  // PublicizeToAtom uses enable_shared_from_this, which means the previous statement, creating a shared_ptr
369  // must occur before the following statement, lest throw std::bad_weak_ptr
370  inserted_bond->publicizeToAtom();
371  }
372 
373  newmol.rwmol.reset(new RDKit::RWMol(*rwmol));
374 
375  return newmol;

384  iConformer Molecule::generateIConformerForGivenRDKitConformerID(unsigned int id,
385  bool centroidNormalization) const {
386  iConformer conformer;
387  RDKit::Conformer rdkit_conformer = this->rwmol->getConformer(id);
388 
389  unsigned int num_atoms = this->atoms.size();
390  conformer.x.reserve(static_cast<size_t>(num_atoms));
391  conformer.y.reserve(static_cast<size_t>(num_atoms));
392  conformer.z.reserve(static_cast<size_t>(num_atoms));
393  conformer.type.reserve(static_cast<size_t>(num_atoms));
394 
395  conformer.num_rotatable_bond = this->numberOfRotatableBonds;
396 
397 
398  using namespace boost::accumulators;
399  accumulator_set<double, stats<tag::mean, tag::count, tag::min, tag::max> > acc_x, acc_y, acc_z;
400 
401  if(centroidNormalization) {
402  for (auto atom_it = rwmol->beginAtoms(); atom_it != rwmol->endAtoms(); ++atom_it) {
403  const RDGeom::Point3D &position = rdkit_conformer.getAtomPos((*atom_it)->getIdx());
404  acc_x(position.x);
405  acc_y(position.y);
406  acc_z(position.z);
407  }
408 
409  conformer.centroidNormalizingTransform.x = mean(acc_x);
410  conformer.centroidNormalizingTransform.y = mean(acc_y);
411  conformer.centroidNormalizingTransform.z = mean(acc_z);
412 
413  }else {
414  conformer.centroidNormalizingTransform.x = 0.0;
415  conformer.centroidNormalizingTransform.y = 0.0;
416  conformer.centroidNormalizingTransform.z = 0.0;
417  }
418 
419  for (auto atom_it = rwmol->beginAtoms(); atom_it != rwmol->endAtoms(); ++atom_it) {
420 
421  // Find the corresponding Atom in this->atoms
422  auto current_Atom_it = (std::find_if(std::begin(this->atoms), std::end(this->atoms),
423  [&](const std::shared_ptr<Atom> &e) {
424  return e->getAtomID() == (*atom_it)->getIdx();
425  }));
426  if (current_Atom_it == std::end(this->atoms)) {
427  BOOST_LOG_TRIVIAL(error) << "Cannot find current atom";
428  std::exit(3);
429  }
430  std::shared_ptr<Atom> current_Atom = *current_Atom_it;
431  conformer.type.push_back(static_cast<unsigned char>((*atom_it)->getAtomicNum()));
432  conformer.variant.push_back((unsigned int) current_Atom->variant);
433 
434  const RDGeom::Point3D &position = rdkit_conformer.getAtomPos((*atom_it)->getIdx());
435 
436  if(centroidNormalization) {
437  conformer.x.push_back(position.x - conformer.centroidNormalizingTransform.x );
438  conformer.y.push_back(position.y - conformer.centroidNormalizingTransform.y );
439  conformer.z.push_back(position.z - conformer.centroidNormalizingTransform.z );
440  }else{
441 
442  conformer.x.push_back(position.x);
443  conformer.y.push_back(position.y);
444  conformer.z.push_back(position.z);
445  }
446 
447  conformer.atomicRadius.push_back(atomTypeToAtomicRadius(
448  stringToAtomType(boost::to_upper_copy<std::string>((*atom_it)->getSymbol()))
449  ));
450  }
451 
452  return conformer;
453  }
